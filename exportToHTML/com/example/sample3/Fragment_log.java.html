package com.example.sample3;

import android.app.AlertDialog;
import android.app.Dialog;
import android.os.Bundle;
import android.util.Log;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.Button;

import androidx.fragment.app.DialogFragment;

public class Fragment_log extends DialogFragment {
    @Override
    public Dialog onCreateDialog(Bundle savedInstanceState) {
        AlertDialog.Builder builder = new AlertDialog.Builder(requireActivity());
        LayoutInflater inflater = requireActivity().getLayoutInflater();
        View view = inflater.inflate(R.layout.fragment_log, null);
        builder.setView(view);

        Button yesButton = view.findViewById(R.id.button_ok);
        Button noButton = view.findViewById(R.id.button_no);

        yesButton.setOnClickListener(v -> {
            // 本当に削除する処理をここに書く
            Log.d("Fragment_log", "はいが押されました");

            // 例：フラグメントを閉じる、削除処理実行など
            requireActivity().getSupportFragmentManager().beginTransaction()
                    .remove(this)
                    .commit();

            //新しいページに移動
            FragmentB_memo newFragment = new FragmentB_memo(); // 新しいインスタンスを作る
            requireActivity()
                    .getSupportFragmentManager()
                    .beginTransaction()
                    .replace(R.id.fragment_container_memo1, newFragment) // container はフラグメントを表示する FrameLayout の ID
                    .commit();
        });

        noButton.setOnClickListener(v -> {
            // キャンセル（フラグメントを閉じるなど）
            Log.d("Fragment_log", "いいえが押されました");
            requireActivity().getSupportFragmentManager().beginTransaction()
                    .remove(this)
                    .commit();
        });

        return builder.create();
    }

    @Override
    public void onStart() {
        super.onStart();
        if (getDialog() != null && getDialog().getWindow() != null) {
            getDialog().getWindow().setLayout(ViewGroup.LayoutParams.MATCH_PARENT, ViewGroup.LayoutParams.WRAP_CONTENT);
        }
    }

}
