package com.example.sample3;

import android.os.Bundle;

import androidx.fragment.app.Fragment;

import android.util.Log;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.TextView;


public class FragmentA_memo extends Fragment {

    @Override
    public View onCreateView(LayoutInflater inflater, ViewGroup container,
                             Bundle savedInstanceState) {
        // Inflate the layout for this fragment
        return inflater.inflate(R.layout.fragment_a_memoedit, container, false);
    }
    @Override
    public void onViewCreated(View view, Bundle savedInstanceState) {
        super.onViewCreated(view, savedInstanceState);

        // view から TextView を取得
        TextView tv = view.findViewById(R.id.textView_memoedit);

        if (tv != null) {
            tv.setOnClickListener(v -> {
                getParentFragmentManager().beginTransaction()
                        .setReorderingAllowed(true)
                        .replace(R.id.fragment_container_memo1, new FragmentB_memo())
                        .addToBackStack(null)
                        .commit();
            });
        } else {
            Log.e("FragmentA_memo", "myTextViewA が見つかりません");
        }
    }

}
